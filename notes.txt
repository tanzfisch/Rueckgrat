# install docker

sudo apt update
sudo apt install extrepo
sudo extrepo enable docker-ce
sudo apt update
sudo apt install docker-ce docker-ce-cli containerd.io docker-buildx-plugin docker-compose-plugin

sudo usermod -aG docker $USER

# install dependencies for llama.cpp
sudo apt install git build-essential cmake curl libopenblas-dev -y -qq

- for CUDA support
sudo apt install -y nvidia-cuda-toolkit

- for ROCm support
sudo apt install -y rocm-dev rocm-hip-runtime rocblas hipblas

- for vulkan support
sudo apt install -y mesa-vulkan-drivers vulkan-tools libvulkan-dev glslc
